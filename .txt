import React, { useState } from 'react';
import { FaAngleDown, FaFilter, FaMobile, FaBoxOpen, FaUsers, FaTags } from 'react-icons/fa';

const Sidebar = () => {
  const [isBrand, setIsBrand] = useState(true);
  const [isAvailability, setIsAvailability] = useState(true);
  const [isGender, setIsGender] = useState(true);
  const [isType, setIsType] = useState(true);
  
  const [selectedFilters, setSelectedFilters] = useState({
    brands: [],
    availability: [],
    gender: [],
    type: []
  });

  const handleFilterChange = (category, value) => {
    setSelectedFilters(prev => ({
      ...prev,
      [category]: prev[category].includes(value)
        ? prev[category].filter(item => item !== value)
        : [...prev[category], value]
    }));
  };

  const FilterSection = ({ title, icon: Icon, isOpen, toggle, children }) => (
    <div className="bg-white rounded-xl shadow-sm border border-gray-100 overflow-hidden hover:shadow-md transition-all duration-300">
      <button
        onClick={toggle}
        className="w-full p-4 flex items-center justify-between hover:bg-gray-50 transition-colors duration-200"
      >
        <div className="flex items-center gap-3">
          <div className="w-8 h-8 bg-gradient-to-br from-blue-500 to-purple-600 rounded-lg flex items-center justify-center text-white">
            <Icon className="text-sm" />
          </div>
          <span className="font-semibold text-gray-800">{title}</span>
        </div>
        <FaAngleDown 
          className={`text-gray-400 transition-transform duration-300 ${isOpen ? 'rotate-180' : ''}`}
        />
      </button>
      
      <div className={`overflow-hidden transition-all duration-500 ease-out ${
        isOpen ? 'max-h-96 pb-4' : 'max-h-0'
      }`}>
        <div className="px-4 space-y-2 max-h-60 overflow-y-auto scrollbar-thin scrollbar-thumb-gray-300 scrollbar-track-gray-100">
          {children}
        </div>
      </div>
    </div>
  );

  const CheckboxItem = ({ label, category, value, count }) => {
    const isSelected = selectedFilters[category]?.includes(value);
    
    return (
      <label className="flex items-center justify-between p-2 rounded-lg hover:bg-blue-50 cursor-pointer transition-colors duration-200 group">
        <div className="flex items-center gap-3">
          <div className="relative">
            <input
              type="checkbox"
              checked={isSelected}
              onChange={() => handleFilterChange(category, value)}
              className="sr-only"
            />
            <div className={`w-4 h-4 rounded border-2 flex items-center justify-center transition-all duration-200 ${
              isSelected 
                ? 'bg-blue-500 border-blue-500' 
                : 'border-gray-300 group-hover:border-blue-400'
            }`}>
              {isSelected && (
                <svg className="w-2.5 h-2.5 text-white" fill="currentColor" viewBox="0 0 20 20">
                  <path fillRule="evenodd" d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z" clipRule="evenodd" />
                </svg>
              )}
            </div>
          </div>
          <span className="text-gray-700 font-medium text-sm group-hover:text-gray-900">{label}</span>
        </div>
        {count && (
          <span className="text-xs bg-gray-100 text-gray-600 px-2 py-1 rounded-full">
            {count}
          </span>
        )}
      </label>
    );
  };

  const brands = [
    { label: 'Samsung', value: 'samsung', count: '124' },
    { label: 'Apple', value: 'apple', count: '89' },
    { label: 'Infinix', value: 'infinix', count: '67' },
    { label: 'Xiaomi', value: 'xiaomi', count: '45' },
    { label: 'OnePlus', value: 'oneplus', count: '32' }
  ];

  const availability = [
    { label: 'In Stock', value: 'in-stock', count: '234' },
    { label: 'Available', value: 'available', count: '189' },
    { label: 'Out of Stock', value: 'out-of-stock', count: '8' }
  ];

  const gender = [
    { label: 'Men', value: 'men', count: '156' },
    { label: 'Women', value: 'women', count: '142' },
    { label: 'Unisex', value: 'unisex', count: '89' }
  ];

  const type = [
    { label: 'Leather', value: 'leather', count: '78' },
    { label: 'Transparent', value: 'transparent', count: '65' },
    { label: 'Silicone', value: 'silicone', count: '112' },
    { label: 'Metal', value: 'metal', count: '34' }
  ];

  const totalSelected = Object.values(selectedFilters).reduce((acc, arr) => acc + arr.length, 0);

  return (
    <aside className="w-80 bg-gradient-to-br from-gray-50 to-blue-50 p-6 min-h-screen">
      <div className="space-y-6">
        {/* Header */}
        <div className="bg-white rounded-2xl p-6 shadow-sm border border-gray-100">
          <div className="flex items-center gap-4">
            <div className="w-12 h-12 bg-gradient-to-br from-blue-500 to-purple-600 rounded-xl flex items-center justify-center text-white shadow-lg">
              <FaFilter className="text-lg" />
            </div>
            <div>
              <h2 className="text-xl font-bold text-gray-800">Filters</h2>
              <p className="text-sm text-gray-500">
                {totalSelected > 0 ? `${totalSelected} filters applied` : 'Refine your search'}
              </p>
            </div>
          </div>
        </div>

        {/* Filter Sections */}
        <div className="space-y-4">
          <FilterSection
            title="Brands"
            icon={FaMobile}
            isOpen={isBrand}
            toggle={() => setIsBrand(!isBrand)}
          >
            {brands.map((brand) => (
              <CheckboxItem
                key={brand.value}
                label={brand.label}
                category="brands"
                value={brand.value}
                count={brand.count}
              />
            ))}
          </FilterSection>

          <FilterSection
            title="Availability"
            icon={FaBoxOpen}
            isOpen={isAvailability}
            toggle={() => setIsAvailability(!isAvailability)}
          >
            {availability.map((item) => (
              <CheckboxItem
                key={item.value}
                label={item.label}
                category="availability"
                value={item.value}
                count={item.count}
              />
            ))}
          </FilterSection>

          <FilterSection
            title="Gender"
            icon={FaUsers}
            isOpen={isGender}
            toggle={() => setIsGender(!isGender)}
          >
            {gender.map((item) => (
              <CheckboxItem
                key={item.value}
                label={item.label}
                category="gender"
                value={item.value}
                count={item.count}
              />
            ))}
          </FilterSection>

          <FilterSection
            title="Type"
            icon={FaTags}
            isOpen={isType}
            toggle={() => setIsType(!isType)}
          >
            {type.map((item) => (
              <CheckboxItem
                key={item.value}
                label={item.label}
                category="type"
                value={item.value}
                count={item.count}
              />
            ))}
          </FilterSection>
        </div>

        {/* Clear All Button */}
        {totalSelected > 0 && (
          <div className="bg-white rounded-xl p-4 shadow-sm border border-gray-100">
            <button
              onClick={() => setSelectedFilters({ brands: [], availability: [], gender: [], type: [] })}
              className="w-full py-3 px-4 bg-gradient-to-r from-red-500 to-pink-500 text-white font-semibold rounded-lg hover:from-red-600 hover:to-pink-600 transition-all duration-200 shadow-sm hover:shadow-md"
            >
              Clear All Filters ({totalSelected})
            </button>
          </div>
        )}
      </div>

      <style jsx>{`
        .scrollbar-thin::-webkit-scrollbar {
          width: 4px;
        }
        .scrollbar-track-gray-100::-webkit-scrollbar-track {
          background-color: #f3f4f6;
          border-radius: 2px;
        }
        .scrollbar-thumb-gray-300::-webkit-scrollbar-thumb {
          background-color: #d1d5db;
          border-radius: 2px;
        }
        .scrollbar-thumb-gray-300::-webkit-scrollbar-thumb:hover {
          background-color: #9ca3af;
        }
      `}</style>
    </aside>
  );
};

export default Sidebar;
